歐幾里德算法／輾轉相除法

操作：對於兩數aaa、bbb，可在O(log(min(a,b)))O(log(min(a, b)))O(log(min(a,b)))內找到最大公因數

觀察最大公因數，可發現以下特性：

  • gcd(a,b)=gcd(b,a)gcd(a, b) = gcd(b, a)gcd(a,b)=gcd(b,a)
  • gcd(a,b)=gcd(∣a∣,∣b∣)gcd(a, b) = gcd(|a|, |b|)gcd(a,b)=gcd(∣a∣,∣b∣)
  • gcd(a,0)=agcd(a, 0) = agcd(a,0)=a
  • gcd(a,b)=gcd(b mod a,a)gcd(a, b) = gcd(b\ mod\ a, a)gcd(a,b)=gcd(b mod a,a)

因此可用以下算法求得最大公因數：

  • 預處理：
    • 對兩數a1a_1a1​、b1b_1b1​做絕對值
    • 若a1>b1a_1 > b_1a1​>b1​，交換兩數（保證a1≤b1a_1 \leq b_1a1​≤b1​）
  • 遞迴：
    • 若ai=0a_i = 0ai​=0，答案為bib_ibi​，否則繼續下一步
    • gcd(ai+1,bi+1)=gcd(bi mod ai,ai)gcd(a_{i + 1}, b_{i + 1}) = gcd(b_i\ mod\ a_i, a_i)gcd(ai+1​,bi+1​)=gcd(bi​ mod ai​,ai​)，回到上一步檢查

歐幾里德算法的程式碼：

// 確保a, b為正，a <= b
int gcd(int a, int b) {
    if (a == 0)
        return b;
    return gcd(b % a, a); 
}

擴展歐幾里德算法

邏輯見貝祖公式證明。

擴展歐幾里德的程式碼：

// ax + by = gcd(a, b)
// 求x, y的解
ll exgcd(ll a, ll b, ll &x, ll &y) {
    // 最後一層遞歸，a = 0時結束遞迴
    if (a == 0) {
        x = 0;
        y = 1;
        return b;
    }

    ll ret = exgcd(b % a, a, x, y);
    ll temp = x;
    // 推導結果
    x = -(b / a) * x + y;
    y = temp;

    return ret;
}

模逆元

a−1×a≡1(mod n)a^{-1} \times a \equiv 1(mod\ n)a−1×a≡1(mod n)

  • a−1a^{-1}a−1為aaa對nnn的模逆元，一般用於除法取膜的運算
  • aaa、nnn互質才存在模逆元
  • 可對貝祖公式，bbb代nnn、模數代nnn求aaa對nnn的模逆元：
    a×x1+n×y1=1a \times x_1 + n \times y_1 = 1a×x1​+n×y1​=1
    a×x1≡1(mod n)a \times x_1 \equiv 1(mod\ n)a×x1​≡1(mod n)
    x1≡a−1x_1 \equiv a^{-1}x1​≡a−1

模逆元的程式碼：

// 擴展歐幾里德 
ll exgcd(ll a, ll b, ll &x, ll &y) {
    if (a == 0) {
        x = 0;
        y = 1;
        return b;
    }

    ll ret = exgcd(b % a, a, x, y);
    ll temp = x;
    x = -(b / a) * x + y;
    y = temp;

    return ret;
}

// 求i對模數p的模逆元
ll invMod(ll i, ll p) {
    i = abs(i);
    p = abs(p);

    ll x, y, ret;
    // a代(i % p)，b代p，保證a <= p
    ret = exgcd(i % p, p, x, y);
    // 取最小正整數
    return (ret == 1) ? ((x % p) + p) % p : -1;
}
